Why Neoluma?

It's a modern programming language designed with developer experience in mind.

Lightning Fast

Compiles to native machine code via LLVM backend for maximum performance

JS-like Syntax

Clean, readable syntax inspired by JS and Python with C#/C++ architecture benefits

Type Safety

Statically typed with optional type inference for robust, error-free code

Smart Memory Management

Both automatic and manual memory management options for different use cases

Language packs

Allows developers to import packages from other languages, and run them natively

Cross-Platform

Write once, run everywhere - from mobile apps to operating system kernels

// to write it correctly.
fn main() {
    // Initialize SDL
    if (SDL.init(SDL.INIT_VIDEO) != 0) {
        print("SDL failed to initialize: ${SDL.GetError()}")
        return;
    }

    // Create a window
    window = SDL.createWindow(
        "Hello Neoluma + SDL3",
        SDL.WINDOWPOS_CENTERED,
        SDL.WINDOWPOS_CENTERED,
        800,
        600,
        SDL.WINDOW_SHOWN
    )

    if (window == null) {
        print("Failed to create window: ${SDL.GetError()}")
        SDL.quit()
        return;
    }

    // Main loop
    running = true
    event = SDL.Event()

    while (running) {
        while (SDL.PollEvent(&event) != 0) {
            if (event.type == SDL.QUIT) {
                running = false
            }
        }

        // Set draw color and clear
        SDL.SetRenderDrawColor(window, 50, 75, 100, 255)
        SDL.RenderClear(window)

        // Render stuff here...

        SDL.RenderPresent(window)
    }

    // Cleanup
    SDL.DestroyWindow(window)
    SDL.quit()
}

#baremetal
#import "memory" as Memory

@entry
fn _start() {
    vga = Memory.map(0xb8000, (80*25)); // map VGA buffer
    message = "Hello, Neoluma Baremetal!";
    for (i : message.length()) (
        vga.write(i, message[i], 0x0F); // 0x0F = white on black
    )
    while (true) {}
}
